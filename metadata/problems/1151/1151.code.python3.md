class Solution:
    def minSwaps(self, data: List[int]) -> int:
        n = len(data)
        m = sum(data)
        ans = m
        s = [0]
        for i in data:
            s.append(s[-1] + i)
        for i in range(m, n + 1):
            ans = min(ans, m - (s[i] - s[i - m]))
        return ans
