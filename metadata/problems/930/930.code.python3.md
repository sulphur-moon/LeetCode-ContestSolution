class Solution:
    def numSubarraysWithSum(self, A: List[int], S: int) -> int:
        s = 0
        f = [0] * (len(A) + 1)
        f[0] = 1
        ans = 0
        for i in A:
            s += i
            if s >= S:
                ans += f[s - S]
            f[s] += 1
        return ans
