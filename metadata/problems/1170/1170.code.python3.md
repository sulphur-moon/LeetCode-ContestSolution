class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(s):
            c = collections.Counter(s)
            k = min(c.keys())
            return c[k]
        wc = []
        for w in words:
            wc.append(f(w))
        ans = []
        for q in queries:
            cnt = 0
            qf = f(q)
            for i in wc:
                if i > qf:
                    cnt += 1
            ans.append(cnt)
        return ans
