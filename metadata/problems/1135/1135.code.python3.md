class DSU(object):
    def __init__(self, n):
        self.par = [i for i in range(n)]
        self.sz = [1] * n

    def find(self, x):
        if self.par[x] != x:
            self.par[x] = self.find(self.par[x])
        return self.par[x]

    def union(self, x, y):
        xr, yr = self.find(x), self.find(y)
        if xr == yr:
            return
        if self.sz[xr] < self.sz[yr]:
            xr, yr = yr, xr
        self.par[yr] = xr
        self.sz[xr] += self.sz[yr]
        
class Solution:
    def minimumCost(self, N: int, conections: List[List[int]]) -> int:
        en = len(conections)
        if en < N - 1:
            return -1
        # 边按权值排序
        conections.sort(key=lambda x:x[2])
        ans, cnt = 0, 0
        # 并查集检查连通性
        dsu = DSU(N)
        for x, y, w in conections:
            # 如果两端点不连通，则将这条边加入
            if dsu.find(x - 1) != dsu.find(y - 1):
                ans += w
                cnt += 1
                # 如果加到 N - 1 条边，就满足最小生成树
                if cnt == N - 1:
                    return ans
                dsu.union(x - 1, y - 1)
        return -1
