class Solution:
    def numberOfDays(self, Y: int, M: int) -> int:
        big = {1, 3, 5, 7, 8, 10, 12}
        if M == 2:
            if (Y % 4) == 0:
                if (Y % 100) == 0:
                    if (Y % 400) == 0:
                        return 29
                    else:
                        return 28
                else:
                    return 29
            return 28
        else:
            if M in big:
                return 31
            else:
                return 30
