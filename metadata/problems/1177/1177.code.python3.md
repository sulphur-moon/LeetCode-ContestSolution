class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        ans = []
        n = len(s)
        cnt = [[0] * 26 for _ in range(n + 1)]
        for i in range(n):
            for j in range(26):
                cnt[i + 1][j] = cnt[i][j]
            cnt[i + 1][ord(s[i]) - 97] += 1
        for left, right, k in queries:
            odd = 0
            for i in range(26):
                if (cnt[right + 1][i] - cnt[left][i]) % 2 == 1:
                    odd += 1
            ans.append(k >= odd // 2)
        return ans
