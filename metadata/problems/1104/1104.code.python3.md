class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        if label == 1:
            return [1]
        ans = []
        while label > 0:
            ans.append(label)
            label //= 2
        ans = ans[::-1]
        base = 2
        n = len(ans)
        flg = n % 2
        for i in range(1, n):
            if i % 2 == flg:
                s = base * 3 - 1
                ans[i] = s - ans[i]
            base *= 2
        return ans
